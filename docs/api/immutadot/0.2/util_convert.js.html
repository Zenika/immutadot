util/convert.js

  1. import lodashFpConvert from './lodashFpConvert'
  2. import update from '../object/update'
  4. /**
  5.  * Wraps an <code>updater</code> function, returning a new function taking <code>object</code>, <code>path</code> and <code>…args</code> as parameters.<br/>
  6.  * The <code>updater</code> function is invoked with <code>value</code> and <code>…args</code>.<br/>
  7.  * Be carefull, the <code>updater</code> function must not mutate its <code>value</code> argument.
  8.  * @function
  9.  * @memberof util
  10.  * @param {function} updater The updater function.
  11.  * @return {function} Returns the wrapped function.
  12.  * @example <caption>Wrapping an updater</caption>
  13.  * const inc = (v, i = 1) => v + i // this function increments a number with an optional value which defaults to 1
  14.  * const incProp = convert(inc)
  15.  * const object = { nested: { prop: 4 } }
  16.  * incProp(object, 'nested.prop') // => { nested: { prop: 5 } }
  17.  * incProp(object, 'nested.prop', 2) // => { nested: { prop: 6 } }
  18.  * @see {@link object.update|update} for more information.
  19.  * @since 0.2.0
  20.  */
  21. const convert = updater => (obj, path, ...rest) => update(obj, path, updater, ...rest)
  22. export default convert
  24. /**
  25.  * Converts and wraps a lodash/fp function.
  26.  * @function
  27.  * @memberof util
  28.  * @param {function} fn The lodash/fp function.
  29.  * @return {function} Returns the wrapped function.
  30.  * @see {@link util.convert|convert} for more information.
  31.  * @since 0.2.0
  32.  * @private
  33.  */
  34. export const convertLodashFp = fn => convert(lodashFpConvert(fn))
