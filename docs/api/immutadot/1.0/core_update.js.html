  1. import { apply } from 'path/apply'
  3. const updateOperation = (obj, prop, value, updater, ...args) => { obj[prop] = updater(value, ...args) }
  5. /**
  6.  * Updates the value at <code>path</code> of <code>object</code> using the <code>updater</code> function.<br/>
  7.  * The updater is invoked with <code>value</code> and <code>…args</code>.<br/>
  8.  * Be carefull, the <code>updater</code> function must not mutate its <code>value</code> argument.
  9.  * @function
  10.  * @memberof core
  11.  * @param {Object} obj The object to modify.
  12.  * @param {Array|string} path The path of the property to set.
  13.  * @param {function} updater The function to produce the updated value.
  14.  * @param {...*} args The remaining args.
  15.  * @return {Object} Returns the updated object.
  16.  * @example <caption>Updating a prop</caption>
  17.  * const inc = (v, i = 1) => v + i // this function increments a number with an optional value which defaults to 1
  18.  * const object = { nested: { prop: 4 } }
  19.  * update(object, 'nested.prop', inc) // => { nested: { prop: 5 } }
  20.  * update(object, 'nested.prop', inc, 2) // => { nested: { prop: 6 } }
  21.  * @since 1.0.0
  22.  */
  23. const update = apply(updateOperation)
  25. export { update }
