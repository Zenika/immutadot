  7. /**
  8.  * Internal version of <code>update</code> without support of <code>updater</code> only.
  9.  * @memberof core
  10.  * @param {Object} object The object to modify.
  11.  * @param {Array|string} path The path of the property to set.
  12.  * @param {function} updater The function to produce the updated value.
  13.  * @param {...*} args The remaining args.
  14.  * @return {Object} Returns the updated object.
  15.  * @since 0.1.5
  16.  * @private
  17.  */
  18. export const updatePassingArgs = (object, path, updater, ...args) => rawUpdate(object, path, v => updater(v, ...args))
  20. /**
  21.  * Updates the value at <code>path</code> of <code>object</code> using the <code>updater</code> function.<br/>
  22.  * The updater is invoked with <code>value</code> and <code>…args</code>.<br/>
  23.  * <code>update</code> can be called with <code>updater</code> only, returning a function taking only <code>object</code>, <code>path</code> and <code>…args</code>.
  24.  * @function update
  25.  * @memberof core
  26.  * @param {Object} [object] The object to modify.
  27.  * @param {Array|string} [path] The path of the property to set.
  28.  * @param {function} updater The function to produce the updated value.
  29.  * @param {...*} args The remaining args.
  30.  * @return {Object|function} Returns the updated object or the wrapped function.
  31.  * @example <caption>Updating a prop</caption>
  32.  * const inc = (v, i = 1) => v + i // this function increments a number with an optional value which defaults to 1
  33.  * const object = { nested: { prop: 4 } }
  34.  * update(object, 'nested.prop', inc) // => { nested: { prop: 5 } }
  35.  * update(object, 'nested.prop', inc, 2) // => { nested: { prop: 6 } }
  36.  * @example <caption>Wrapping an updater</caption>
  37.  * const inc = (v, i = 1) => v + i // this function increments a number with an optional value which defaults to 1
  38.  * const incProp = update(inc)
  39.  * const object = { nested: { prop: 4 } }
  40.  * incProp(object, 'nested.prop') // => { nested: { prop: 5 } }
  41.  * incProp(object, 'nested.prop', 2) // => { nested: { prop: 6 } }
  42.  * @see {@link https://lodash.com/docs#update|lodash.update} for more information.
  43.  * @since 0.1.5
  44.  */
  45. export default (...args) => {
  46.   if (args.length === 1 && isFunction(args[0]))
  47.     return (obj, path, ...rest) => updatePassingArgs(obj, path, args[0], ...rest)
  48.   return updatePassingArgs(...args)
  49. }
