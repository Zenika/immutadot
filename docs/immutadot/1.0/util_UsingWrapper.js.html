  15.  * Instances are created by calling {@link util.using}.
  16.  * @memberof util
  17.  * @see {@link util.using} for more information.
  18.  * @private
  19.  * @since 0.1.12
  20.  */
  21. class UsingWrapper {
  23.   /**
  24.    * This constructor should not be called directly.<br/>
  25.    * Instances are created by calling {@link util.using}.
  26.    * @param {...(Array|string)} paths The paths to use as arguments.
  27.    * @see {@link util.using} for more information.
  28.    * @since 0.1.12
  29.    */
  30.   constructor(...paths) {
  31.     this._paths = paths
  32.   }
  34.   /**
  35.    * Argument placeholder.
  36.    * @since 0.2.1
  37.    */
  38.   static placeholder = Symbol.for('immutadot.using.placeholder')
  40.   /**
  41.    * Call a function with the specified arguments and possibly some more arguments.
  42.    * @param {function} fn The function to call.
  43.    * @param {Object} object The object to modify.
  44.    * @param {Array|string} path The path of the property to be set.
  45.    * @param {Array} pArgs The arguments for the function call.
  46.    * @return {Object} Returns the updated object.
  47.    * @since 0.1.12
  48.    */
  49.   _call(fn, object, path, pArgs) {
  50.     let callArgs = pArgs
  51.     const args = this._paths.map(usingPath => {
  52.       if (isSymbol(usingPath)) {
  53.         const arg = head(callArgs)
  54.         callArgs = callArgs.slice(1)
  55.         return arg
  56.       }
  57.       return get(object, usingPath)
  58.     }).concat(callArgs)
  59.     return fn(object, path, ...args)
  60.   }
  61. }
  63. // Add namespaces functions to the UsingWrapper prototype
  64. const { convert, unset, toPath, ...filteredCore } = core
  65. const namespaces = [
  66.   array,
  67.   filteredCore,
  68.   lang,
  69.   object,
  70.   string,
  71. ]
  72. namespaces.forEach(namespace => {
  73.   for (const fnName in namespace) {
  74.     const fn = namespace[fnName]
  75.     UsingWrapper.prototype[fnName] = function(object, path, ...args) {
  76.       return this._call(fn, object, path, args)
  77.     }
  78.   }
  79. })
