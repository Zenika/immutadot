  1. import { unsafeToPath } from 'path/toPath'
  3. /**
  4.  * Wrapper allowing to make sequences of immutadot functions calls on an object.<br/>
  5.  * Instances are created by calling {@link seq.chain}.<br/>
  6.  * The result of such sequences must be unwrapped with {@link seq.ChainWrapper#value}.
  7.  * @memberof seq
  8.  * @see {@link seq.chain} for more information.
  9.  * @private
  10.  * @since 0.1.8
  11.  */
  12. class ChainWrapper {
  14.   /**
  15.    * This constructor should not be called directly.<br/>
  16.    * Instances are created by calling {@link seq.chain}.
  17.    * @param {Object} wrapped The object to wrap.
  18.    * @param {Array|string} [path] The path of the object on which functions are called.
  19.    * @param {Array} [pFlow=[]] Current calls flow.
  20.    * @see {@link seq.chain} for more information.
  21.    * @since 0.1.8
  22.    */
  23.   constructor(wrapped, path, pFlow = []) {
  24.     this._wrapped = wrapped
  25.     this._path = path
  26.     this._flow = pFlow
  27.     this._commited = null
  28.   }
  30.   /**
  31.    * Translates a relative path to an absolute path, using <code>this._path</code> as a base path.
  32.    * @param {Array|string} path A relative path.
  33.    * @return {Array} Absolute path.
  34.    * @since 0.1.11
  35.    */
  36.   _absolutePath(path) {
  37.     return unsafeToPath(this._path).concat(unsafeToPath(path))
  38.   }
  40.   /**
  41.    * Add a function call to the sequence.
  42.    * @param {function} fn The function to call.
  43.    * @param {Array|string} path The path of the property to be set.
  44.    * @param {...*} args The arguments for the function call.
  45.    * @returns {seq.ChainWrapper} The new wrapper instance.
  46.    * @since 0.1.8
  47.    */
  48.   _call(fn, path, args) {
  49.     return new ChainWrapper(
  50.       this._wrapped,
  51.       this._path,
  52.       [...this._flow, object => fn(object, this._absolutePath(path), ...args)],
  53.     )
  54.   }
  56.   /**
  57.    * Executes the chain sequence.
  58.    * @returns {seq.ChainWrapper} The new wrapper instance.
  59.    * @since 0.3.0
  60.    */
  61.   commit() {
  62.     if (this._flow.length === 0) return this
  64.     if (this._commited === null) {
  65.       this._commited = new ChainWrapper(
  66.         this._flow.reduce((obj, fn) => fn(obj), this._wrapped),
  67.         this._path,
  68.       )
  69.     }
  71.     return this._commited
  72.   }
  74.   /**
  75.    * Function to be called by {@link seq.peek|peek} with the resolved unwrapped value.
  76.    * @memberof seq
  77.    * @callback peekCallback
  78.    * @param {Object} unwrapped The resolved unwrapped object
  79.    * @since 0.3.0
  80.    */
  82.   /**
  83.    * Executes the chain sequence and calls <code>callback</code> with the unwrapped object.
  84.    * @param {seq.peekCallback} callback Function to be called with the resolved unwrapped value.
  85.    * @returns {seq.ChainWrapper} The new wrapper instance.
  86.    * @todo Add an example.
  87.    * @since 0.3.0
  88.    */
  89.   peek(callback) {
  90.     const commited = this.commit()
  92.     callback(commited._wrapped)
  94.     return commited
  95.   }
  97.   /**
  98.    * Executes the chain sequence and returns the unwrapped object.
  99.    * @returns {Object} Returns the resolved unwrapped object.
  100.    * @example
  101.    * chain({ nested1: { prop: 'old' }, nested2: { prop: 1 } })
  102.    *   .set('nested1.prop', 'new')
  103.    *   .unset('nested2.prop')
  104.    *   .value() // => { nested1: { prop: 'new' }, nested2: {} }
  105.    * @see {@link seq.chain|chain} for more information.
  106.    * @since 0.1.8
  107.    */
  108.   value() {
  109.     return this.commit()._wrapped
  110.   }
  111. }
  113. export { ChainWrapper }
