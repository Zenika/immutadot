  7. import concat from 'lodash/concat'
  8. import flow from 'lodash/flow'
  9. import mapValues from 'lodash/mapValues'
  10. import toPath from 'lodash/toPath'
  12. /**
  13.  * Wrapper allowing to make sequences of immutadot functions calls on an object.<br/>
  14.  * Instances are created by calling {@link seq.chain}.<br/>
  15.  * The result of such sequences must be unwrapped with {@link seq.ChainWrapper#value}.
  16.  * @memberof seq
  17.  * @see {@link seq.chain} for more information.
  18.  * @private
  19.  * @since 0.1.8
  20.  */
  21. class ChainWrapper {
  23.   /**
  24.    * This constructor should not be called directly.<br/>
  25.    * Instances are created by calling {@link seq.chain}.
  26.    * @param {Object} wrapped The object to wrap.
  27.    * @param {Array|string} [path] The path of the object on which functions are called.
  28.    * @param {Array} [pFlow=[]] Current calls flow.
  29.    * @see {@link seq.chain} for more information.
  30.    * @since 0.1.8
  31.    */
  32.   constructor(wrapped, path, pFlow = []) {
  33.     this._wrapped = wrapped
  34.     this._path = path
  35.     this._flow = pFlow
  36.     this._commited = null
  37.   }
  39.   /**
  40.    * Translates a relative path to an absolute path, using <code>this._path</code> as a base path.
  41.    * @param {Array|string} path A relative path.
  42.    * @return {Array} Absolute path.
  43.    * @since 0.1.11
  44.    */
  45.   _absolutePath(path) {
  46.     return concat(toPath(this._path), toPath(path))
  47.   }
  49.   /**
  50.    * Add a function call to the sequence.
  51.    * @param {function} fn The function to call.
  52.    * @param {Array|string} path The path of the property to be set.
  53.    * @param {...*} args The arguments for the function call.
  54.    * @returns {seq.ChainWrapper} The new wrapper instance.
  55.    * @since 0.1.8
  56.    */
  57.   _call(fn, path, args) {
  58.     return new ChainWrapper(
  59.       this._wrapped,
  60.       this._path,
  61.       concat(this._flow, object => fn(object, this._absolutePath(path), ...args)),
  62.     )
  63.   }
  65.   /**
  66.    * Executes the chain sequence.
  67.    * @returns {seq.ChainWrapper} The new wrapper instance.
  68.    * @since 0.3.0
  69.    */
  70.   commit() {
  71.     if (this._flow.length === 0) return this
  73.     if (this._commited === null)
  74.       this._commited = new ChainWrapper(
  75.         flow(this._flow)(this._wrapped),
  76.         this._path,
  77.       )
  79.     return this._commited
  80.   }
  82.   /**
  83.    * Executes the chain sequence and calls <code>callback</code> with the unwrapped object.
  84.    * @param {seq.peekCallback} callback Function to be called with the resolved unwrapped value.
  85.    * @returns {seq.ChainWrapper} The new wrapper instance.
  86.    * @todo Add an example.
  87.    * @since 0.3.0
  88.    */
  89.   peek(callback) {
  90.     const commited = this.commit()
  92.     callback(commited._wrapped)
  94.     return commited
  95.   }
  97.   /**
  98.    * Executes the chain sequence and returns the unwrapped object.
  99.    * @returns {Object} Returns the resolved unwrapped object.
  100.    * @example
  101.    * chain({ nested1: { prop: 'old' }, nested2: { prop: 1 } })
  102.    *   .set('nested1.prop', 'new')
  103.    *   .unset('nested2.prop')
  104.    *   .value() // => { nested1: { prop: 'new' }, nested2: {} }
  105.    * @see {@link seq.chain|chain} for more information.
  106.    * @since 0.1.8
  107.    */
  108.   value() {
  109.     return this.commit()._wrapped
  110.   }
  111. }
  113. /**
  114.  * Function to be called by {@link seq.peek|peek} with the resolved unwrapped value.
  115.  * @memberof seq
  116.  * @callback peekCallback
  117.  * @param {Object} unwrapped The resolved unwrapped object
  118.  * @since 0.3.0
  119.  */
  121. // Add namespaces functions to the ChainWrapper prototype
  122. [
  123.   array,
  124.   collection,
  125.   lang,
  126.   math,
  127.   object,
  128. ].forEach(namespace => Object.assign(
  129.   ChainWrapper.prototype,
  130.   mapValues(
  131.     namespace,
  132.     fn => function(path, ...args) {
  133.       return this._call(fn, path, args) // eslint-disable-line no-invalid-this
  134.     },
  135.   ),
  136. ))
  138. export { ChainWrapper, ChainWrapper as default }
