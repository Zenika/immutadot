  46.    * @since 0.3.0
  47.    */

  49.     this.chainWrapperRef.chainWrapper = this.chainWrapperRef.chainWrapper.set(concat(this.path, property), value)
  50.     return true
  51.   }

  59.    */
  60.   deleteProperty(target, property) {
  61.     this.chainWrapperRef.chainWrapper = this.chainWrapperRef.chainWrapper.unset(concat(this.path, property))
  62.     return true
  63.   }

  68.    * @since 0.3.0
  69.    */
  70.   _peek() {
  71.     let peeked
  72.     this.chainWrapperRef.chainWrapper = this.chainWrapperRef.chainWrapper.peek(_peeked => { peeked = _peeked })
  73.     return isEmpty(this.path) ? peeked : get(peeked, this.path)
  74.   }
  75. }

  92.   if (!isObject(reference)) throw TypeError('Cannot protect a non-object value')
  93.   return callback => {
  94.     const chainWrapper = chain(reference)

  96.     const proxy = new Proxy(reference, new ProtectHandler(chainWrapperRef))
  97.     callback(proxy)
  98.     return chainWrapperRef.chainWrapper.value()
  99.   }
  100. }
