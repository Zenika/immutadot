  1. import * as array from '../array'
  2. import * as collection from '../collection'

  4. import * as lang from '../lang'
  5. import * as math from '../math'
  6. import * as object from '../object'
  8. import at from 'lodash/at'
  9. import concat from 'lodash/concat'
  10. import map from 'lodash/map'
  11. import mapValues from 'lodash/mapValues'
  12. import omit from 'lodash/omit'
  13. import toPath from 'lodash/toPath'
  14. import { updatePassingArgs } from './update'
  16. /**
  17.  * Wrapper allowing to specify one or several paths to use as arguments for an immutadot function call.<br/>
  18.  * Instances are created by calling {@link core.using}.
  19.  * @memberof core
  20.  * @see {@link core.using} for more information.
  21.  * @private
  22.  * @since 0.1.12
  23.  */
  24. class UsingWrapper {
  26.   /**
  27.    * This constructor should not be called directly.<br/>
  28.    * Instances are created by calling {@link core.using}.
  29.    * @param {...(Array|string)} paths The paths to use as arguments.
  30.    * @see {@link core.using} for more information.
  31.    * @since 0.1.12
  32.    */
  33.   constructor(...paths) {
  34.     this._paths = map(paths, toPath)
  35.   }
  37.   /**
  38.    * Call a function with the specified arguments and possibly some more arguments.
  39.    * @param {function} fn The function to call.
  40.    * @param {Object} object The object to modify.
  41.    * @param {Array|string} path The path of the property to be set.
  42.    * @param {...*} args The arguments for the function call.
  43.    * @return {Object} Returns the updated object.
  44.    * @since 0.1.12
  45.    */
  46.   _call(fn, object, path, args) {
  47.     return fn(object, path, ...concat(at(object, this._paths), args))
  48.   }
  49. }
  51. // Add namespaces functions to the UsingWrapper prototype
  52. [
  53.   array,
  54.   collection,
  55.   {
  56.     ...omit(core, ['unset']), // Unset doesn't take any parameters
  57.     update: updatePassingArgs, // Avoid updater only version of update
  58.   },
  59.   lang,
  60.   math,
  61.   object,
  62. ].forEach(namespace => Object.assign(
  63.   UsingWrapper.prototype,
  64.   mapValues(
  65.     namespace,
  66.     fn => function(object, path, ...args) {
  67.       return this._call(fn, object, path, args) // eslint-disable-line no-invalid-this
  68.     }
  69.   )
  70. ))
  72. export default UsingWrapper
